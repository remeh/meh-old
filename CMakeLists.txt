cmake_minimum_required(VERSION 3.1.0)

project(meh)

set(CMAKE_AUTOMOC ON)
set(CMAKE_AUTORCC ON)
set(CMAKE_AUTOUIC ON)

if(CMAKE_VERSION VERSION_LESS "3.7.0")
    set(CMAKE_INCLUDE_CURRENT_DIR ON)
endif()

find_package(Qt6 COMPONENTS Network REQUIRED)

    window.cpp

if (APPLE)
    set(MACOSX_BUNDLE_BUNDLE_NAME meh)
    set(MACOSX_BUNDLE_GUI_IDENTIFIER "io.remy.meh")
    set(MACOSX_BUNDLE_LONG_VERSION_STRING ${PROJECT_VERSION})
    set(MACOSX_BUNDLE_SHORT_VERSION_STRING ${PROJECT_VERSION})
    set(MACOSX_BUNDLE_BUNDLE_VERSION ${PROJECT_VERSION})
    set(MACOSX_BUNDLE_ICON_FILE icon.icns)
    set(app_icon_macos "${CMAKE_CURRENT_SOURCE_DIR}/res/icon.icns")
    set_source_files_properties(${app_icon_macos} PROPERTIES MACOSX_PACKAGE_LOCATION "Resources")

    add_executable(meh MACOSX_BUNDLE ${SOURCES} ${app_icon_macos})
else()
    add_executable(meh ${SOURCES})
endif()

target_link_libraries(meh Qt6::Widgets)
target_link_libraries(meh Qt6::Network)
